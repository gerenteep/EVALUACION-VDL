// ═══════════════════════════════════════════════════════════ // APPS SCRIPT — Sistema Evaluación VDL // Pega este código en Google Apps Script y reimplementa // ═══════════════════════════════════════════════════════════ const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { const callback = e.parameter.callback; const action = e.parameter.action; let result; try { if (action === 'read') { result = leer(e); } else { result = {ok: false, msg: 'Acción no válida'}; } } catch(err) { result = {ok: false, msg: err.toString()}; } // JSONP para evitar CORS desde archivos locales o GitHub Pages const json = JSON.stringify(result); if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); let result; if (data.action === 'write') result = escribir(data); else if (data.action === 'readAll') result = leerTodo(data); else result = {ok: false, msg: 'Acción desconocida'}; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, msg: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function escribir(data) { const hoja = obtenerHoja('Datos'); const fila = [ new Date().toISOString(), data.cargo, data.asesor, data.mes, data.evaluador, data.semana, data.indId, data.entregable, data.meta, data.logrado, data.cumpl, data.nota, data.obs ]; const rows = hoja.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][1] === data.cargo && rows[i][2] === data.asesor && rows[i][3] === data.mes && rows[i][4] === data.evaluador && rows[i][5] === data.semana && rows[i][6] === data.indId) { hoja.getRange(i + 1, 1, 1, fila.length).setValues([fila]); return {ok: true, msg: 'Actualizado'}; } } hoja.appendRow(fila); return {ok: true, msg: 'Guardado'}; } function leer(e) { const hoja = obtenerHoja('Datos'); const rows = hoja.getDataRange().getValues(); const cargo=e.parameter.cargo, asesor=e.parameter.asesor, mes=e.parameter.mes, evaluador=e.parameter.evaluador, semana=e.parameter.semana; const data = rows.slice(1) .filter(r => r[1]===cargo && r[2]===asesor && r[3]===mes && r[4]===evaluador && r[5]===semana) .map(r => ({indId:r[6], meta:r[8], logrado:r[9], cumpl:r[10], nota:r[11], obs:r[12]})); return {ok: true, data: data}; } function leerTodo(data) { const hoja = obtenerHoja('Datos'); const rows = hoja.getDataRange().getValues(); const result = rows.slice(1) .filter(r => r[1]===data.cargo && r[2]===data.asesor && r[3]===data.mes) .map(r => ({evaluador:r[4], semana:r[5], indId:r[6], meta:r[8], logrado:r[9], cumpl:r[10], nota:r[11], obs:r[12]})); return {ok: true, data: result}; } function obtenerHoja(nombre) { let h = SS.getSheetByName(nombre); if (!h) { h = SS.insertSheet(nombre); h.appendRow(['Timestamp','Cargo','Asesor','Mes','Evaluador','Semana', 'IndID','Entregable','Meta','Logrado','Cumpl%','Nota','Observaciones']); h.getRange(1,1,1,13) .setFontWeight('bold') .setBackground('#1A3C6E') .setFontColor('#FFFFFF'); h.setFrozenRows(1); } return h; }
